# 题目：
　　从尾到头打印链表

## 题目描述：
　　输入一个链表，从尾到头打印链表每个节点的值。

## 解题：
```c++
/**
*  struct ListNode {
*        int val;
*        struct ListNode *next;
*        ListNode(int x) :
*              val(x), next(NULL) {
*        }
*  };
*/
class Solution {
public:
    vector<int> printListFromTailToHead(struct ListNode* head) {
        vector<int> v;
        stack<int> s;
        if(head == NULL) return v;
        struct ListNode* t = head;
        while(t){
            s.push(t->val);
            t = t->next;
        }
        while(!s.empty()){
            int a = s.top();
            v.push_back(a);
            s.pop();
        }
        return v;
    }
};
```
